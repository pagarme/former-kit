```jsx
const IconWarning = require('emblematic-icons/svg/Warning32.svg').default;

<Alert
  type="warning"
  icon={<IconWarning height={16} width={16} />}
>
  <p><strong>Warning.</strong> Something is going on!</p>
</Alert>
```
